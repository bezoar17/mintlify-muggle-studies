class Solution {
public:
        
vector<int> getRow(int rowIndex) {

	 vector<int> r1;r1.push_back(1);
	 if(rowIndex==0) return r1;
	 vector<int> r2;
	 for(int i=0;i<=rowIndex;i++){
	 	r2.clear();
	 	r2.push_back(1);
	 	for(int j=1;j<i;j++)
	 		r2.push_back(r1[j]+r1[j-1]);
	 	r2.push_back(1);
	 	r1.swap(r2);
	 }
	 return r1;   
    }
};
