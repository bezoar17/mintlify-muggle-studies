# @param {Character[][]} board
# @return {Boolean}
def is_valid_sudoku(board)
  h_row = Array.new(9) {Set.new}; h_col = Array.new(9) {Set.new}; 
  h_box = Hash.new {|h,k| h[k]=Set.new}
  
  (0...9).each do |i|
    (0...9).each do |j|
      value = board[i][j]
      next if value == "."; box_idx = (i/3).to_s + (j/3).to_s;
      return false if h_row[i].include?(value) || h_col[j].include?(value) || h_box[box_idx].include?(value)
      h_row[i] << value; h_col[j] << value; h_box[box_idx] << value
    end
  end
  true
end
