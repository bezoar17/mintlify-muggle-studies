      Simple no. of islands like problem, find connected components using dfs, each time you start a new island(connected component), start a counter and return the max size of an island
    </Card>

```ruby title="Max Area of Island" lines
# @param {Integer[][]} grid
# @return {Integer}
def max_area_of_island(grid)
  r,c = grid.size, grid[0].size
  # visited = 2, island = 1, water = 0
  max_island_size = 0
  @island_count = 0

  (0...r).each do |i|
    (0...c).each do |j|
      if grid[i][j] == 1
        @island_count = 0
        dfs(grid, i, j)
        max_island_size = [@island_count, max_island_size].max
      end
    end
  end
  max_island_size
end

def dfs(grid, i, j)
  grid[i][j] = 2; @island_count += 1
  directions = [[1,0],[-1,0],[0,1],[0,-1]]

  for dx, dy in directions do
    x = i+dx; y= j+dy
    if x.between?(0, grid.size-1) && y.between?(0, grid[0].size-1) && grid[x][y] == 1
      dfs(grid, x, y)
    end
  end
end
```
