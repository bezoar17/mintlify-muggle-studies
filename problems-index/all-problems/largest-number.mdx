class Solution {
public:
    static bool comp(const int &a,const int &b){
        long long int aa=stol(to_string(a)+to_string(b));
        long long int bb=stol(to_string(b)+to_string(a));
        return aa > bb;
    }
    string largestNumber(vector<int>& nums) {
        sort(nums.begin(),nums.end(),comp);
        string s;
        for(int i=0;i<nums.size();i++)
        s+=to_string(nums[i]);
        s.erase(1, s.find_first_not_of('0'));
        return s;
    }
};
