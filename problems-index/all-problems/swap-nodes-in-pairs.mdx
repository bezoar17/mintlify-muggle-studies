/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* swapPairs(ListNode* A) {
        ListNode* current=A;
        if(A==NULL || A->next==NULL )
          return A;
    
        int count=0;int temp_val;
        while(current!=NULL)
        {
          count++;
          if(count%2!=0 && current->next!=NULL) // every odd node ( counting starts from 1) except if it is the last one.
          {
            temp_val=current->val;
            current->val=current->next->val;
            current->next->val=temp_val;
          }
          current=current->next;
        }
        return A;
        }
};
