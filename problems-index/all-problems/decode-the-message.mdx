```ruby title="Decode the Message" lines
# @param {String} key
# @param {String} message
# @return {String}
def decode_message(key, message)
  cipher = {}
  key.each_char do |ch|
    next if ch == ' '
    cipher[ch] = (cipher.size + 'a'.ord).chr if cipher[ch].nil?
  end
  cipher[' ']= ' '

  message.chars.map { |ch| cipher[ch] }.join
end
```
