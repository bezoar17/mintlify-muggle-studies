  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Due to constraints, just simulate the 4 directional walk. Backtracking to do dfs, by the end all empty cells must be visited.  For visited it will change 0 to -2 and do a dfs in 4 direction
</Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Unique Paths III" lines
# @param {Integer[][]} grid
# @return {Integer}
def unique_paths_iii(grid)
  m = grid.size; n = grid[0].size;
  empty_cells = 1; start_i = start_j = nil
  result = 0;

  (0...m).each do |i|
    (0...n).each do |j|
      start_i,start_j = i,j if grid[i][j]==1
      empty_cells +=1 if grid[i][j]==0
    end
  end

  dfs=->(i,j, current_empty) {
    return unless i.between?(0,m-1) && j.between?(0,n-1) && grid[i][j] >= 0
    return result+=(current_empty==0 ? 1:0) if grid[i][j]==2

    grid[i][j]=-2
    dfs.call(i+1, j, current_empty-1)
    dfs.call(i-1, j, current_empty-1)
    dfs.call(i, j+1, current_empty-1)
    dfs.call(i, j-1, current_empty-1)
    grid[i][j]=0
  }
  dfs.call(start_i, start_j, empty_cells)
  result
end
```
