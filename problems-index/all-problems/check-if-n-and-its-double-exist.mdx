# @param {Integer[]} arr
# @return {Boolean}
def check_if_exist(arr)
  seen = Set.new

  arr.each do |n|
    return true if seen.include?(n*2) || (n.even? && seen.include?(n/2))
    seen.add(n)
  end
  false
end
