    <Card>
      Floyd Warshall, and then just pick node which satisfies this condition
    </Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Find the City With the Smallest Number of Neighbors at a Threshold Distance" lines
# @param {Integer} n
# @param {Integer[][]} edges
# @param {Integer} distance_threshold
# @return {Integer}
def find_the_city(n, edges, distance_threshold)
  dist = Array.new(n) { Array.new(n, Float::INFINITY) }
  n.times { |i| dist[i][i] = 0 }
  
  edges.each { |u, v, w| dist[u][v] = dist[v][u] = w }
  
  n.times { |k| n.times { |i| n.times { |j| dist[i][j] = [dist[i][j], dist[i][k] + dist[k][j]].min } } }
  
  dist.each_with_index
      .min_by { |row, i| [row.count { |d| d <= distance_threshold }, -i] }
      .last
end
```
