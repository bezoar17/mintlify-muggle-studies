```ruby title="Number of Flowers in Full Bloom" lines
# @param {Integer[][]} flowers
# @param {Integer[]} people
# @return {Integer[]}
def full_bloom_flowers(flowers, people)
  bloom_start = flowers.map(&:first).sort
  bloom_end = flowers.map(&:last).map{|x| x+1}.sort
  n = bloom_start.size
  
  people.map { |time|
    start_time = bloom_start.bsearch_index{|x| x>time} || n
    end_time = bloom_end.bsearch_index{|x| x>time} || n
    start_time-end_time
  }
end
```
