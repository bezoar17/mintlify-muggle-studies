  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      All cells around a water cell have to have a height of 1, then cells around 1 have to have 2, and so on.
Multi-source bfs starting from water cells.
</Card>

```ruby title="Map of Highest Peak" lines
# @param {Integer[][]} is_water
# @return {Integer[][]}
def highest_peak(is_water)
  m = is_water.size; n = is_water[0].size
  # initially 0-land, 1-water
  # multisource bfs from water cells, to use same input as final result and visited array
  # we can update water to 0, land to -1, and -1 would indicate unvisited cell
  # height for neighbor is 1+current_cell, and water already has 0
  
  queue = []
  (0...m).each do |i|
    (0...n ).each do |j|
      is_water[i][j] == 1 ? (is_water[i][j]=0; queue<<[i,j]) : (is_water[i][j]=-1)
    end
  end
  directions = [[0,1],[0,-1],[1,0],[-1,0]]
  neighbors=->(x,y) {
    directions.map{|dx,dy|
      ((x+dx).between?(0,m-1) && (y+dy).between?(0,n-1) && is_water[x+dx][y+dy]==-1) ? [x+dx, y+dy] : nil
    }.compact
  }

  until queue.empty?
    level_cells = queue
    queue = []

    level_cells.each do |i,j|
      neighbors.call(i,j).each do |n_i, n_j|
        is_water[n_i][n_j] = is_water[i][j]+1
        queue << [n_i,n_j]
      end
    end
  end

  is_water
end
```
