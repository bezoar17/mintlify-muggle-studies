# @param {Integer[]} coins
# @param {Integer} amount
# @return {Integer}
def coin_change(coins, amount)
  coins_sorted = coins.sort.reverse
  min_coins_for_amount = Array.new(amount+1)
  min_coins_for_amount[0] = 0
  
  coins_sorted.select{|x| x<=amount}.each { |coin| min_coins_for_amount[coin] = 1 }

  (1..amount).each do |target|
    min_coins_for_amount[target] = coins_sorted.map do |coin|
        next unless (target-coin).between?(0, amount)
        next if min_coins_for_amount[target - coin].nil?
        1 + min_coins_for_amount[target - coin]
    end.compact.min
  end

  min_coins_for_amount[amount] || -1
end
