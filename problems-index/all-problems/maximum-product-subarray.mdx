# @param {Integer[]} nums
# @return {Integer}
def max_product(nums)
  # for 0, split and get ans
  # for even negatives, multiply all
  # for odd negatives,
  n = nums.size

  prefix = suffix = 1
  res = -Float::INFINITY

  (0...n).each do |idx|
    prefix = 1 if prefix == 0
    suffix = 1 if suffix == 0

    prefix = prefix * nums[idx]
    suffix = suffix * nums[n-1-idx]
    res = [res, prefix, suffix].max
  end
  res
end
