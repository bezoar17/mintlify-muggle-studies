class Solution {
public:
    void recur(vector<string> &ans,string digits,int index,string current,char hash[][4]){
        
        if(digits.size()==0)
        return;
   		if(index==digits.size()){
   		    for(int i=0;i<current.size();i++)
   		    if(current.at(i)=='\x00')
   		    return;
   			ans.push_back(current);
   			return;
   		}
   		for(int i=0;i<4;i++){
   			string temp=current;
   			current+=hash[digits.at(index)-'0'][i];
   			recur(ans,digits,index+1,current,hash);
   			current=temp;
   		}
   	}	
	vector<string> letterCombinations(string digits) {
        
       vector <string> ret;
       char  phone[10][4]={ {' ',  '{${code}$}', '{${code}$}', '{${code}$}' }, //0
                         {'{${code}$}', '{${code}$}', '{${code}$}', '{${code}$}' }, //1
                         {'a',  'b',  'c',  '{${code}$}' }, //2
                         {'d',  'e',  'f',  '{${code}$}' }, //3
                         {'g',  'h',  'i',  '{${code}$}' }, //4
                         {'j',  'k',  'l',  '{${code}$}' }, //5
                         {'m',  'n',  'o',  '{${code}$}' }, //6
                         {'p',  'q',  'r',  's'  }, //7
                         {'t',  'u',  'v',  '{${code}$}' }, //8
                         {'w',  'x',  'y',  'z'  }  //9
                       };

        recur(ret,digits,0,"",phone);
        return ret;
    }
};
