```ruby title="Queries on Number of Points Inside a Circle" lines
# @param {Integer[][]} points
# @param {Integer[][]} queries
# @return {Integer[]}
def count_points(points, queries)
    queries.map do |x, y, r|
        points.count do |p,q|
           (p-x)**2 + (q-y)**2 <= r*r
        end
    end
end
```
