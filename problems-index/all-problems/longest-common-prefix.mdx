# @param {String[]} strs
# @return {String}
def longest_common_prefix(strs)
    strs_in_chars = strs.map{ |e| e.chars }
    lowest_length = strs.map(&:size).min
    return "" if lowest_length < 1
    return strs[0] if strs.size == 1
    
    (0...lowest_length).each do |i|
        ch = strs_in_chars[0][i]
        strs_in_chars.each do |strch|
           return strs_in_chars[0][0...i].join("") if strch[i] != ch
        end
    end
    return strs_in_chars[0][0...lowest_length].join("")
end
