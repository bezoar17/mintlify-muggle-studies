```ruby title="Count the Number of Ideal Arrays" lines
def comb(n, k)
  res = 1
  k.times do |i|
    res = res * (n - i) / (i + 1)
  end
  res
end

def ideal_arrays(n, max_value)
  mod = 1_000_000_007
  ans = max_value
  freq = Hash.new(0)
  (1..max_value).each { |x| freq[x] = 1 }

  (1...n).each do |k|
    temp = Hash.new(0)
    freq.each do |x, f|
      m = 2
      while m * x <= max_value
        ans = (ans + comb(n - 1, k) * f) % mod
        temp[m * x] += f
        m += 1
      end
    end
    freq = temp
  end

  ans
end

puts ideal_arrays(2, 5)  # Output: 10
```
