# @param {String[]} recipes
# @param {String[][]} ingredients
# @param {String[]} supplies
# @return {String[]}
def find_all_recipes(recipes, ingredients, supplies)
  available = Set.new(supplies)
  dependency_graph = Hash.new { |h, k| h[k] = [] }
  recipe_to_index = recipes.each_with_index.to_h
  in_degree = Array.new(recipes.size, 0)

  ingredients.each_with_index do |ingredient_list, index|
    ingredient_list.each do |ingredient|
      unless available.include?(ingredient)
        dependency_graph[ingredient] << recipes[index]
        in_degree[index] += 1
      end
    end
  end

  queue = []
  in_degree.each_with_index do |degree, index|
    queue << recipes[index] if degree == 0
  end
  result = []

  until queue.empty?
    current_recipe = queue.shift
    result << current_recipe

    dependency_graph[current_recipe].each do |dependent_recipe|
      recipe_index = recipe_to_index[dependent_recipe]
      in_degree[recipe_index] -= 1
      queue << dependent_recipe if in_degree[recipe_index] == 0
    end
  end

  result
end
