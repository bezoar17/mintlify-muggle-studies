```ruby title="Integer Replacement" lines
# @param {Integer} n
# @return {Integer}
def integer_replacement(n)
  n.between?(1,3) ? n - 1 : 1 + (n % 2 == 0 ? integer_replacement(n / 2) : [integer_replacement(n + 1), integer_replacement(n - 1)].min)
end
```
