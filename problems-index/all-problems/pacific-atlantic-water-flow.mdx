# @param {Integer[][]} heights
# @return {Integer[][]}
def pacific_atlantic(heights)
  m,n = heights.size, heights[0].size
  pacific = Set.new
  atlantic = Set.new

  in_range=->(r,c) { r.between?(0,m-1) && c.between?(0,n-1) }

  dfs=->(row, column, visited, previous_height) {
    return if heights[row][column] < previous_height
    
    return if visited.include?([row, column])
    visited.add([row, column])

    [[-1,0],[1,0],[0,-1],[0,1]].each do |x, y|
      row_x = row + x
      column_y = column + y
      if in_range.call(row_x, column_y)
        dfs.call(row_x, column_y, visited, heights[row][column])
      end
    end
  }

  (0...n).each do |column|
    dfs.call(0, column, pacific, 0)
    dfs.call(m-1, column, atlantic, 0)
  end

  (0...m).each do |row|
    dfs.call(row, 0, pacific, 0)
    dfs.call(row, n-1, atlantic, 0)
  end

  (pacific & atlantic).to_a
end
