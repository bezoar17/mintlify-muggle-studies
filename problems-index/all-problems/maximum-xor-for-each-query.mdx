# @param {Integer[]} nums
# @param {Integer} maximum_bit
def get_maximum_xor(nums, maximum_bit)
    max_no = 2**maximum_bit -1
    curr_val = 0
    nums.map do |elem|
        curr_val = curr_val ^ elem
        max_no ^ curr_val
    end.reverse
end
