```ruby title="Merge Triplets to Form Target Triplet" lines
# @param {Integer[][]} triplets
# @param {Integer[]} target
# @return {Boolean}
def merge_triplets(triplets, target)
  return true if triplets.include?(target)

  filtered = triplets.reject { |t| t[0] > target[0] || t[1] > target[1] || t[2] > target[2] || target.intersection(t).empty? }
  # elements which satisfy the above condition can never be part of the merge operation or final set
  # if we have each target element in one of the sets, then we can reach the solution, as for each column
  # all other sets have lower value than that target

  presence = Array.new(3, false)

  filtered.each { |t| (0..2).each { |i| presence[i] ||= (t[i] == target[i]) } }
  
  presence.all?(true)
end
```
