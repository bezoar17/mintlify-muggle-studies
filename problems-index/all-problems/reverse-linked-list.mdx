```ruby title="Reverse Linked List" lines
# Definition for singly-linked list.
# class ListNode
#     attr_accessor :val, :next
#     def initialize(val = 0, _next = nil)
#         @val = val
#         @next = _next
#     end
# end
# @param {ListNode} head
# @return {ListNode}
def reverse_list(head)
  current_node = head
  return nil if current_node.nil?
  prev_node = nil
  
  while true
    next_node = current_node.next # save info for intial next node

    current_node.next = prev_node # switch current nodes pointer

    prev_node = current_node # move the current and prev nodes to the next one in original
    current_node = next_node
    break if current_node.nil?
  end

  prev_node
end
```
