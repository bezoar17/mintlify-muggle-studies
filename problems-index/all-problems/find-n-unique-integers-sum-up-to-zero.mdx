defmodule Solution do
  @spec sum_zero(n :: integer) :: [integer]
  def sum_zero(0), do: []
  def sum_zero(n) when rem(n,2) == 0, do: [n,-n] ++ sum_zero(n-2)
  def sum_zero(n) when rem(n,2) == 1, do: [0] ++ sum_zero(n-1)
