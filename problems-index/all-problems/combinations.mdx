      <p>Given two integers <code>n</code> and <code>k</code>, return <em>all possible combinations of</em> <code>k</code> <em>numbers chosen from the range</em> <code>[1, n]</code>.</p>  <p>You may return the answer in <strong>any order</strong>.</p>   <p><strong class="example">Example 1:</strong></p>   <strong>Input:</strong> n = 4, k = 2 <br/> <strong>Output:</strong> [[1,2],[1,3],[1,4],[2,3],[2,4],[3,4]] <br/> <strong>Explanation:</strong> There are 4 choose 2 = 6 total combinations. Note that combinations are unordered, i.e., [1,2] and [2,1] are considered to be the same combination.   <p><strong class="example">Example 2:</strong></p>   <strong>Input:</strong> n = 1, k = 1 <br/> <strong>Output:</strong> [[1]] <br/> <strong>Explanation:</strong> There is 1 choose 1 = 1 total combination.    <p><strong>Constraints:</strong></p>  <ul>  <li><code>1 &lt;= n &lt;= 20</code></li>  <li><code>1 &lt;= k &lt;= n</code></li> </ul>

class Solution {
public:
    void comb(vector<vector <int> > &val,int n,int k,int valindex,int index,vector<int> &data){

		if(index==k){
			val.push_back(data);
			return;
		}
		if(valindex>=n+1)
			return;
		comb(val,n,k,valindex+1,index,data);
		data[index]=valindex;
		comb(val,n,k,valindex+1,index+1,data);
	}
   	vector<vector<int>> combine(int n, int k) {
        
        vector<vector <int> > ret;
        vector<int> data(k,0);
        comb(ret,n,k,1,0,data);
        return ret;

    }
};
