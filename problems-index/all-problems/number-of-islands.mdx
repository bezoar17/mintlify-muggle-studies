```ruby title="Number of Islands" lines
# @param {Character[][]} grid
# @return {Integer}
def num_islands(grid)
  # VISITED = 2; ISLAND = 1; WATER = 0
  island_count = 0
  r,c = grid.size, grid[0].size

  (0...r).each do |i|
    (0...c).each do |j|
      if grid[i][j] == "1"
        island_count+=1
        dfs(grid, i, j)
      end
    end
  end
  island_count
end

def dfs(grid, i, j)
  grid[i][j] = "2"
  directions = [[1,0],[-1,0],[0,1],[0,-1]]

  for dx, dy in directions do
    x = i+dx; y= j+dy
    if x.between?(0, grid.size-1) && y.between?(0, grid[0].size-1) && grid[x][y] == "1"
      dfs(grid, x, y)
    end
  end
end
```
