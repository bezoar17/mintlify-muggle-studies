  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      simple bfs, but along with avoiding already visited nodes, we also avoid restricted nodes. And the total visited nodes set by the end will be the component size.
</Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Reachable Nodes With Restrictions" lines
# @param {Integer} n
# @param {Integer[][]} edges
# @param {Integer[]} restricted
# @return {Integer}
def reachable_nodes(n, edges, restricted)
  adj_h = Hash.new{|h,k|h[k]=Set.new}; edges.each {|u,v| adj_h[u].add(v);adj_h[v].add(u);}
  restricted_set = restricted.to_set
  queue= Queue.new; visited = Set.new;
  queue << 0;

  until queue.empty?
    node = queue.pop
    visited.add(node)
    adj_h[node].select{|x| !visited.include?(x) && !restricted_set.include?(x)}.each { |neighbor| queue.push(neighbor)}
  end
  visited.size
end
```
