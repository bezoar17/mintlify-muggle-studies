class Solution {
public:
    vector<vector<int>> generate(int numRows) {
       		
       	vector <vector <int> > ret;
       	if(numRows==0) return ret;
       	vector<int> r1;
       	r1.push_back(1);
       	ret.push_back(r1);
       	for(int i=1;i<numRows;i++){
       		r1.clear();
       		r1.push_back(1);
       		for(int j=1;j<i;j++)
       			r1.push_back(ret[i-1][j]+ret[i-1][j-1]);
       		r1.push_back(1);
       		ret.push_back(r1);
       	}
       	return ret;
     }
};
