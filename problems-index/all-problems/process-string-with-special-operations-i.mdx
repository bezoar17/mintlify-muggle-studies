# @param {String} s
# @return {String}
def process_str(s)
  chars = s.chars
  res = []
  chars.each do |ch|
    if ch == '*'
      res.pop
    elsif ch == '#'
      res+=res
    elsif ch == '%'
      res = res.reverse
    else
      res << ch
    end
  end
  res.join
end
