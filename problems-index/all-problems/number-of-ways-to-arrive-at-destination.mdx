```ruby title="Number of Ways to Arrive at Destination" lines
# @param {Integer} n
# @param {Integer[][]} roads
# @return {Integer}
def count_paths(n, roads)
    mod = 1_000_000_007
    adj = Array.new(n) { [] }

    roads.each do |u, v, t|
        adj[u] << [v, t]
        adj[v] << [u, t]
    end

    shortest_time = Array.new(n, Float::INFINITY)
    cnt = Array.new(n, 0)
    pq = [[0, 0]] # [time, node]

    shortest_time[0] = 0
    cnt[0] = 1

    until pq.empty?
        time, node = pq.shift
        next if time > shortest_time[node]

        adj[node].each do |nbr, rtime|
            if time + rtime < shortest_time[nbr]
                shortest_time[nbr] = time + rtime
                cnt[nbr] = cnt[node]
                pq << [shortest_time[nbr], nbr]
            elsif time + rtime == shortest_time[nbr]
                cnt[nbr] = (cnt[nbr] + cnt[node]) % mod
            end
        end
        pq.sort_by!(&:first)
    end

    cnt[-1]
end
```
