class Solution {
public:
    string convertToTitle(int A) {
        string str;
    int i = 0;  // To store current index in str which is result
 
    while (A>0)
    {
        // Find remainder
        int rem = A%26;
 
        // If remainder is 0, then a 'Z' must be there in output
        if (rem==0)
        {
            str= 'Z'+str;
            A = (A/26)-1;
        }
        else // If remainder is non-zero
        {
            str =char(64+rem)+str;
            A = A/26;
        }
    }
    
    
    return str;
    }
};
