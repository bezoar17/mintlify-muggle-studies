      <p>Given an array of <strong>distinct</strong> integers <code>candidates</code> and a target integer <code>target</code>, return <em>a list of all <strong>unique combinations</strong> of </em><code>candidates</code><em> where the chosen numbers sum to </em><code>target</code><em>.</em> You may return the combinations in <strong>any order</strong>.</p>  <p>The <strong>same</strong> number may be chosen from <code>candidates</code> an <strong>unlimited number of times</strong>. # @param {Integer[]} candidates
# @param {Integer} target
# @return {Integer[][]}
def combination_sum(candidates, target)
  @results = []
  
  backtrack = -> (candidates, target, current_subset, i) do
    if target == 0
      @results << current_subset.dup
      return
    end

    return if target < 0 || i == candidates.size
    n = candidates[i]
    backtrack.call(candidates, target-n, current_subset.dup << n, i)
    backtrack.call(candidates, target, current_subset, i+1)
  end

  backtrack.call(candidates.sort, target, [], 0)
  @results
end
