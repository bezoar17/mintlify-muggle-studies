```ruby title="Find the Number of Copy Arrays" lines
# @param {Integer[]} original
# @param {Integer[][]} bounds
# @return {Integer}
def count_arrays(original, bounds)
  # original diff is applied to starting element to get rest of the array
  n = original.size
  elem_range = bounds.first
  
  (1...n).each do |i|
    diff = original[i]-original[i-1]
    curr_range = elem_range.map {|x| x+diff }
    curr_bounds = bounds[i]
    # bounds and curr_range intersection
    elem_range = [[curr_bounds.first, curr_range.first].max, [curr_bounds.last, curr_range.last].min]
    range_size = elem_range.last-elem_range.first + 1
    return 0 if range_size < 1
  end
  elem_range.last-elem_range.first + 1
end
```
