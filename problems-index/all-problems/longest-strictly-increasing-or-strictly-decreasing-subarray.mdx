# @param {Integer[]} nums
# @return {Integer}
def longest_monotonic_subarray(nums)
  (1...nums.size).each.reduce([1,1,1]) { |(len, inc, dec), i| 
    inc = (nums[i]-nums[i-1] > 0) ? inc+1 : 1
    dec = (nums[i]-nums[i-1] < 0) ? dec+1 : 1
    [[inc,dec,len].max , inc, dec]
  }[0]
end
