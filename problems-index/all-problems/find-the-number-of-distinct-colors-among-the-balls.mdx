  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Just simulate, keep track of ball colours in a hash, also keep count of balls for each color in a hash. Whenever it reaches 0, delete from hash.
</Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Find the Number of Distinct Colors Among the Balls" lines
# @param {Integer} limit
# @param {Integer[][]} queries
# @return {Integer[]}
def query_results(limit, queries)
  ball_color = Hash.new
  color_count = Hash.new(0)
  
  queries.map do |ball, color|
    older_ball_color = ball_color[ball]
    
    if older_ball_color.nil?
      ball_color[ball]=color; color_count[color]+=1
    elsif older_ball_color != color
      color_count[older_ball_color]-=1
      ball_color[ball]=color
      color_count[color]+=1
      
      color_count.delete(older_ball_color) if color_count[older_ball_color]==0
    end
    color_count.size
  end
end
```
