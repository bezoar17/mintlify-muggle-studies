```ruby title="Binary Search" lines
# @param {Integer[]} nums
# @param {Integer} target
# @return {Integer}
def search(nums, target)
  bsearch_index(nums, 0, nums.size-1, target)
end

def bsearch_index(arr, left, right, val)
    mid = left + (right-left)/2
    return mid if arr[mid] == val
    return -1 if left>=right
    interval = arr[mid] > val ? [left, mid] : [mid+1, right]
    bsearch_index(arr, *interval, val)
  end
```
