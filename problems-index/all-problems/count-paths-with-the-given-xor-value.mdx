  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      dp[i][j][k], ways to reach i,j with xor value k. Calculate all possible values, k can go up to 16, so keep that in mind while initializing the array.
    </Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Count Paths With the Given XOR Value" lines
# @param {Integer[][]} grid
# @param {Integer} k
# @return {Integer}
def count_paths_with_xor_value(grid, k)
  m = grid.size
  n = grid[0].size
  dp = Array.new(m) { Array.new(n) { Array.new(16,0) } }
  bigmod = 1_000_000_000 + 7
  dp[0][0][grid[0][0]] = 1

  (0...m).each do |i|
    (0...n).each do |j|
      (0...16).each do |xor|
         new_xor = xor^grid[i][j]
         dp[i][j][new_xor] += dp[i-1][j][xor] if i>0 # top cell
         dp[i][j][new_xor] += dp[i][j-1][xor] if j>0 # left cell
      end
    end
  end

  dp[m-1][n-1][k] % bigmod
end
```
