      Go left to right, match ). Then afterwards, go right to left to match remaining ( values.

```ruby title="Check if a Parentheses String Can Be Valid" lines
# @param {String} s
# @param {String} locked
# @return {Boolean}
def can_be_valid(s, locked)
  n = s.size
  return false if n.odd?
  locked = locked.chars
  s = s.chars
  unlocked_count = 0 
  open_bracket_count = 0
  
  # left to right, match ) with previous unlocked and ( chars
  s.each_with_index do |ch, i|
    if locked[i] == '0'
      unlocked_count+=1
    elsif ch == '('
      open_bracket_count+=1
    elsif ch == ')'
      if open_bracket_count > 0
        open_bracket_count-=1
      elsif unlocked_count > 0
        unlocked_count-=1
      else
        return false
      end
    end
  end

  # right to left match remaining ( 
  balance_count = 0
  (n-1).downto(0).each do |i|
    if locked[i] == '0'
      balance_count-=1
      unlocked_count-=1
    elsif s[i] == '('
      balance_count+=1
      open_bracket_count-=1
    elsif s[i] == ')'
      balance_count-=1
    end

    return false if balance_count > 0
    break if unlocked_count == 0 && open_bracket_count == 0
  end

  return open_bracket_count < 1
end
```
