```ruby title="Path Sum II" lines
# Definition for a binary tree node.
# class TreeNode
#     attr_accessor :val, :left, :right
#     def initialize(val = 0, left = nil, right = nil)
#         @val = val
#         @left = left
#         @right = right
#     end
# end
# @param {TreeNode} root
# @param {Integer} target_sum
# @return {Integer[][]}
def path_sum(root, target_sum)
  valid_paths = []
  return [] if root.nil?
  
  recurse=->(node, current_sum, path) {
    current_sum += node.val
    path << node.val
    valid_paths << path if node.left == nil && node.right == nil && current_sum == target_sum
    
    recurse.call(node.left, current_sum, path.dup) if !node.left.nil?
    recurse.call(node.right, current_sum, path.dup) if !node.right.nil?
  }
  recurse.call(root, 0, [])
  valid_paths
end
```
