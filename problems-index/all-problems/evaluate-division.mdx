# @param {String[][]} equations
# @param {Float[]} values
# @param {String[][]} queries
# @return {Float[]}
def calc_equation(equations, values, queries)
  @adj_h = Hash.new{|h,k| h[k]={}}

  (0...equations.size).each do |i|
    u,v = equations[i]
    @adj_h[u][v]=values[i]
    @adj_h[v][u]=1.0/values[i]
  end;

  def bfs(st, ed)
    visited = Set.new([st])
    queue = [[st, 1.0]]

    until queue.empty?
      node, dist = queue.pop
      @adj_h[node].each do |neighbor, weight|
        next if visited.include?(neighbor)
        return weight*dist if neighbor == ed
        queue << [neighbor, weight*dist]
        visited.add(neighbor)
      end
    end
    -1.0
  end

  res = []

  queries.each do |st, ed|
    (res << -1.0; next) if !@adj_h.has_key?(st) || !@adj_h.has_key?(ed)
    (res << @adj_h[st][ed]; next) if !@adj_h[st][ed].nil?

    (res << 1.0; next) if st == ed

    dist = bfs(st,ed)
    res << dist
    next if dist == -1.0
    @adj_h[st][ed] = dist
    @adj_h[ed][st] = 1.0/dist
  end

  res
end
