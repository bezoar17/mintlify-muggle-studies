# @param {Integer[]} nums
# @return {Integer[][]}
def permute(nums)
  return [nums] if nums.size == 1
  return [nums, nums.reverse] if nums.size == 2
  
  results = []
  nums.each_with_index do |n,i|
    permute(nums - [nums[i]]).each { |arr| results << arr.unshift(nums[i]) }
  end
  results
end
