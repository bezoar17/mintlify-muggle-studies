  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Store everything as a hash of hash, nested hash objects.

Hash is a dir, Array is a file, with contents as values.

Get to parent dict of last value, either sub-dir or filename, then use that. To do action based on file or folder.


    </Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Design In-Memory File System" lines
class FileSystem
  def initialize()
    @root = {} # file is a Array, dir is a Hash
    @root[""] = {}
  end

  def ls(path) # list file name, or if path, list files and directory in this path only
    dir = get_parent_lvl_dict(path)
    filename = get_chunks(path).last
    return [filename] if dir[filename].is_a?(Array) # file
    dir[filename].keys.sort # dir
  end

  def mkdir(path)
    create_dirs(path)
  end

  def add_content_to_file(file_path, content)
    dir = get_parent_lvl_dict(file_path)
    filename = get_chunks(file_path).last
    (dir[filename]||=[]).append(content)
  end

  def read_content_from_file(file_path)
    dir = get_parent_lvl_dict(file_path)
    filename = get_chunks(file_path).last
    dir[filename].join
  end

  private
  def get_chunks(path)
    path == "/" ? [""] : path.split("/")
  end

  # given path, get parent dict, path might be for file or dir
  def get_parent_lvl_dict(path)
    get_chunks(path)[..-2].reduce(@root) { |current_dir, folder| current_dir[folder] }
  end

  # create dirs
  def create_dirs(path)
    get_chunks(path).reduce(@root) { |current_dir, folder| current_dir[folder] ||= {} }
  end
end
```
