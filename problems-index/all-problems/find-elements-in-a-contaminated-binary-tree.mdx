  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Simple recursion, restore the tree first, maintain set of idxs present. Then queries are O(1)
    </Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Find Elements in a Contaminated Binary Tree" lines
# Definition for a binary tree node.
# class TreeNode
#     attr_accessor :val, :left, :right
#     def initialize(val = 0, left = nil, right = nil)
#         @val = val
#         @left = left
#         @right = right
#     end
# end
class FindElements
  def initialize(root)
    @idxs = Set.new
    restore(root, 0)
  end


  def restore(node, og)
    return if node.nil?
    node.val = og
    @idxs << og

    restore(node.left, 2*og+1)
    restore(node.right, 2*og+2)
  end

  def find(target)
    @idxs.include?(target) 
  end
end

# Your FindElements object will be instantiated and called as such:
# obj = FindElements.new(root)
# param_1 = obj.find(target)
```
