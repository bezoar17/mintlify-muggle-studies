```ruby title="Uncrossed Lines" lines
# @param {Integer[]} nums1
# @param {Integer[]} nums2
# @return {Integer}
def max_uncrossed_lines(nums1, nums2)
   lcs(nums1, nums2) 
end


def lcs(a,b)
    m,n = a.size, b.size
    dp = Array.new(m+1) { Array.new(n+1,0) }

    (0..m).each do |i|
      (0..n).each do |j|
          if i ==0 || j ==0
              dp[i][j]=0
          elsif a[i-1] == b[j-1]
              dp[i][j] = dp[i-1][j-1] + 1
          else
              dp[i][j] = [dp[i-1][j], dp[i][j-1]].max
          end
        end
    end
    dp[m][n]
end
```
