# @param {String} s
# @return {Integer}
def longest_valid_parentheses(s)
  # longest will always start from (
  # will always end on )
  # (()((()) -> 4
  stack = [-1] # top of stack always has start of valid substring idx - 1
  max_len = 0
  
  s.chars.each_with_index do |ch, idx|
    if ch == '('
      stack << idx # each ( can be start of valid substring
    else
      stack.pop # match the paranthesis, ((), first ( can now be valid starting point
      if stack.empty? # if there is no valid starting point remaining, # update )'s index as last valid substring position -1
        stack << idx # if we get ))), last idx here is valid start -1
      else
        max_len = [max_len, idx-stack.last].max # calculate longest length based on top of stack
      end
    end
  end

  max_len
end
