# Definition for a binary tree node.
# class TreeNode
#     attr_accessor :val, :left, :right
#     def initialize(val)
#         @val = val
#         @left, @right = nil, nil
#     end
# end

# @param {TreeNode} root
# @param {TreeNode} target
# @param {Integer} k
# @return {Integer[]}
class TreeNode
  attr_accessor :parent

  def related_nodes
    [@parent, @left, @right]
  end
end
@start_node = nil

def add_parents_and_find_start_node(node, parent=nil, target)
  return if node.nil?
  @start_node = node if node.val == target.val
  node.parent = parent
  add_parents_and_find_start_node(node.left, node, target)
  add_parents_and_find_start_node(node.right, node, target)
end

def distance_k(root, target, k)
  return [target.val] if k == 0
  add_parents_and_find_start_node(root, nil, target)
  # start bfs from @start_node
  result = []

  visited = Set.new([])
  queue = [[@start_node, 0]]
  while !queue.empty?
    current_node, current_dist = queue.shift
    visited.add(current_node.val)
    current_node.related_nodes.each do |node|
      if !node.nil? && !visited.include?(node.val)
        queue<<[node, current_dist+1]
        result << node.val if current_dist+1 == k
      end
    end
  end
  result
end
