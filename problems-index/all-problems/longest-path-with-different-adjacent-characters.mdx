```ruby title="Longest Path With Different Adjacent Characters" lines
# @param {Integer[]} parent
# @param {String} s
# @return {Integer}
def longest_path(parent, s)
  diameter = 0; chars = s.chars
  adj_h=Hash.new{|h,k|h[k]=[]}; parent[1..].each_with_index {|parent,i| adj_h[parent]<<(i+1); }

  return 1 if parent.size == 1
  
  depth_of_tree=->(root) {
    children = adj_h[root]
    return 1 if children.empty?
    child_depths = children.map{ |ch| val=depth_of_tree.call(ch); (chars[root]!=chars[ch]) ? val:0}.sort
    diameter = [diameter, 1+child_depths.last(2).sum].max
    return 1 + child_depths.last.to_i
  }

  depth_of_tree.call(0)
  return diameter
end
```
