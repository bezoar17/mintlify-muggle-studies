# @param {Integer} n
# @return {Integer[]}
def count_bits(n)
  (0..n).each.map {|x| hamming_weight(x) } 
end

def hamming_weight(n)
  count = 0
  while n > 0
    count += 1 if n & 1 == 1
    n >>= 1
  end
  count
end
