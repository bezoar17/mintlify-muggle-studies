```ruby title="Minimum Limit of Balls in a Bag" lines
# @param {Integer[]} nums
# @param {Integer} max_operations
# @return {Integer}
def minimum_size(nums, max_operations)
  @nums = nums; @max_operations = max_operations
  left = 1; right = nums.max;
  mid = left + (right-left)/2;
  while left<right
    mid = (right+left)/2
    if can_be_max(mid)
      right = mid
    else
      left = mid+1
    end
  end
  left
end

def can_be_max(max_val)
  ops_req = 0
  @nums.select{|x| x>max_val}.each do |num|
    ops_for_num = (num/max_val.to_f).ceil - 1
    ops_req += ops_for_num
  end
  ops_req <= @max_operations
end
```
