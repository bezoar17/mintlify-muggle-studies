# @param {String} s
# @return {Boolean}
def is_valid(s)
  parens_h = {
    "(" => ")",
    "{" => "}",
    "[" => "]"
  }

  stack = []
  
  s.chars.each do |ch|
    if parens_h.keys.include?(ch)
      stack << ch
    else
      ch == parens_h[stack.last] ? stack.pop : (return false)
    end
  end

  stack.empty?
end
