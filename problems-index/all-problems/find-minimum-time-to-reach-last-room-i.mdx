      Djikstra, starting from 0,0 with dist 0. Dist of neighbor of i is 1+[dist_i, move_time[neighbor]].max

```ruby title="Find Minimum Time to Reach Last Room I" lines
# @param {Integer[][]} move_time
# @return {Integer}
def min_time_to_reach(move_time)
  n = move_time.size; m = move_time[0].size
  visited = Array.new(n) { Array.new(m, false) }
  dist = Array.new(n) { Array.new(m, Float::INFINITY) }
  
  dj_pq = Containers::MinHeap.new
  dj_pq << [0,[0,0]]

  neighbors=->(x,y){
    directions = [[0,1],[1,0],[-1,0],[0,-1]]
    directions.map{|dx, dy|
      (x+dx).between?(0,n-1) && (y+dy).between?(0,m-1) && !visited[x+dx][y+dy] ? [x+dx, y+dy] : nil
    }.compact
  }
  
  until dj_pq.empty?
    curr_time, (i,j) = dj_pq.pop
    next if visited[i][j]
    visited[i][j]=true
    dist[i][j]=curr_time
    return curr_time if i == n-1 && j ==m-1

    neighbors.call(i,j).each do |ni, nj|
      neighbor_time = 1 + [curr_time, move_time[ni][nj]].max
      dj_pq.push([neighbor_time, [ni,nj]])
    end
  end
end
```
