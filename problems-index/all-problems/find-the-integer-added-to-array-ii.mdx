# @param {Integer[]} nums1
# @param {Integer[]} nums2
# @return {Integer}
def minimum_added_integer(nums1, nums2)
  # nums1 has exactly 2 more
  # given the constraints, it can be just under n^3
  # to get 2 removals, is n^2, to check if x is valid, we can break when there are 2 diffs
  # so its not always O(n)
  nums1.sort!; nums2.sort!;
  n=nums2.size; n1= n+2;
  x = Float::INFINITY
  
  x_for =->(i,j) {
    idx1=0;idx2=0; res=nil;
    while idx1<n1
      unless idx1==i || idx1==j
        diff = nums2[idx2]-nums1[idx1]
        res = diff if res.nil?
        return Float::INFINITY if res!=diff
        idx2+=1
      end
      idx1+=1
    end
    res
  }

  (0...(n1-1)).each do |i|
    ((i+1)...n1).each do |j|
      x = [x, x_for.call(i,j)].min
    end
  end

  x
end
