/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int symetric(TreeNode *A, TreeNode *B){

    if(!A && B)
        return 0;
    if(!B && A)
        return 0;
    if(!A)
        return 1;
    if(A->val != B->val)
        return 0;
    return (symetric(A->left,B->right) && symetric(A->right,B->left));
    
}
    bool isSymmetric(TreeNode* A) {
        return symetric(A,A);
    }
};
