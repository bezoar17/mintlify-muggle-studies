# @param {Integer[]} height
# @return {Integer}
def trap(height)
  n = height.size
  calc =-> (a,b) { [a,b].max }
  left_max_till_i = [height[0]]
  right_max_till_i = [height[-1]]
  (1...n).each { |j| left_max_till_i << calc.call(left_max_till_i.last, height[j]) }
  (n-2).downto(0).each { |j| right_max_till_i.unshift( calc.call(right_max_till_i.first, height[j]) )}
  
  (0...n).reduce(0) do |res,i|
    # min of (left max till now, right max till now) - current height; ith index will store this volume.
    volume_at_i = [left_max_till_i[i], right_max_till_i[i]].min - height[i]
    res + volume_at_i
  end
end
