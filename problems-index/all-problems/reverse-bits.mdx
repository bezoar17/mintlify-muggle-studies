class Solution {
public:
    uint32_t reverseBits(uint32_t A) {
     unsigned int msb=1;
    unsigned int lsb=1;
    unsigned int ans=0;
    int i;
    for(i=0;i<31;i++)
        msb=msb<<1;
    for(i=0;i<32;i++){
       if(msb&A)
        ans=ans|(lsb);
       lsb=lsb<<1;
       msb=msb>>1;
    }
    return ans;   
    }
};
