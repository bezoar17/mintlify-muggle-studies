      Just simulate change in direction and no. of elements to collect in that direction.
</Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="Spiral Matrix III" lines
# @param {Integer} rows
# @param {Integer} cols
# @param {Integer} r_start
# @param {Integer} c_start
# @return {Integer[][]}
def spiral_matrix_iii(rows, cols, r_start, c_start)
  # Store all possible directions in an array.
  dir = [[0, 1], [1, 0], [0, -1], [-1, 0]]
  traversed = []

  # Initial step size is 1, value of d represents the current direction.
  step = 1
  direction = 0
  while traversed.length < rows * cols
    # direction = 0 -> East, direction = 1 -> South
    # direction = 2 -> West, direction = 3 -> North
    2.times do
      step.times do
        # Validate the current position
        if r_start >= 0 && r_start < rows && c_start >= 0 && c_start < cols
          traversed << [r_start, c_start]
        end
        # Make changes to the current position.
        r_start += dir[direction][0]
        c_start += dir[direction][1]
      end
      direction = (direction + 1) % 4
    end
    step += 1
  end
  traversed
end
```
