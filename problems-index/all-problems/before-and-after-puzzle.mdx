# @param {String[]} phrases
# @return {String[]}
def before_and_after_puzzles(phrases)
  result = []
  phrases_arr = phrases.map(&:split)

  (0...phrases.size).each do |i|
    (0...phrases.size).each do |j|
      p1, p2 = phrases_arr[i], phrases_arr[j]
      next if i==j || p1.last != p2.first
      result << (p1 + p2[1..]).join(" ")
    end
  end
  
  result.uniq.sort
end
