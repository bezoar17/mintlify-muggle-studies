# @param {String} s
# @param {String} t
# @return {Boolean}
def is_anagram(s, t)
  return false if s.size != t.size
  
  h_s = s.chars.reduce(Hash.new(0)){ |h_, ch| h_[ch]+=1; h_ }
  h_t = t.chars.reduce(Hash.new(0)){ |h_, ch| h_[ch]+=1; h_ }

  h_s == h_t
end
