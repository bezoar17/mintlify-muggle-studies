class Solution {
public:
    int countPrimes(int n) {
        if(n<2) return 0;
        vector<bool> marks(n+1,0);
        marks[0]=marks[1]=1;int c=0;
        for(int i=2;i<=sqrt(n);i++){
            if(marks[i]==0)
            c++;
            for(int j=i*i;j<=n;j+=i)
            {
                marks[j]=1;
            }
        }
        for(int i=1+sqrt(n);i<n;i++)
        {
            if(marks[i]==0)
            c++;
        }
        return c;
    }
};
