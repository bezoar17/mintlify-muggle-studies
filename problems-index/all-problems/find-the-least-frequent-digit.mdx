```ruby title="Find The Least Frequent Digit" lines
# @param {Integer} n
# @return {Integer}
def get_least_frequent_digit(n)
  freq = n.to_s.chars.tally
  min_freq = freq.values.min
  freq.select {|k,v| v==min_freq}.map(&:first).min.to_i
end
```
