# @param {Integer[][]} grid
# @return {Integer[]}
def find_missing_and_repeated_values(grid)
  hash, duplicate = {}, nil
  grid.flatten!
  grid.each do |val|
    duplicate = val if hash[val]
    hash[val] = true
  end

  [duplicate, ((grid.size * (grid.size + 1))/2) - (grid.sum - duplicate)]  
end
