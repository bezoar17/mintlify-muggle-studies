# @param {Integer[]} nums
# @return {Integer[][]}
def subsets_with_dup(nums)
  @results = []
  backtrack = ->(nums, current_subset, index) do
    if index == nums.size
      @results << current_subset.dup
      return
    end
    
    current_subset << nums[index]
    backtrack.call(nums, current_subset.dup, index+1)
    current_subset.pop
    backtrack.call(nums, current_subset.dup, index+1)    
  end

  backtrack.call(nums.sort, [], 0)
  @results.uniq
end
