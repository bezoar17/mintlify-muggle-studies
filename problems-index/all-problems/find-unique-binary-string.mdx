# @param {String[]} nums
# @return {String}
def find_different_binary_string(nums)
  present = nums.map {|x| x.to_i(2) }.to_set
  (0..(1<<nums.size)).detect {|x| !present.include?(x) }.to_s(2).rjust(nums.size,'0')
end
