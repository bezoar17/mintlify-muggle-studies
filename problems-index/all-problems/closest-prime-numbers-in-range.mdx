  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Generate all primes and store in global, to be used across test cases. Then, select primes in range, and calculate the diff between 2 adjacent ones, pick the one with minimum diff.
</Card>

```ruby title="Closest Prime Numbers in Range" lines
# @param {Integer} left
# @param {Integer} right
# @return {Integer[]}

def generate_primes(k)
  # Handle edge cases
  return [] if k < 2

  # Create array from 2 to k
  is_prime = Array.new(k + 1, true)
  is_prime[0] = is_prime[1] = false

  # Use Sieve of Eratosthenes algorithm
  (2..Math.sqrt(k)).each do |i|
    if is_prime[i]
      (i * i).step(k, i) { |j| is_prime[j] = false }
    end
  end

  # Collect all prime numbers
  (2..k).select { |num| is_prime[num] }
end
@all_primes = generate_primes(1_000_000)

def closest_primes(left, right)
  p_in_range = @all_primes.select {|x| x>=left && x<=right }
  return [-1, -1] if p_in_range.size < 2
  result = [-1,-1]
  diff = Float::INFINITY

  (1...p_in_range.size).each do |i|
    d = p_in_range[i]-p_in_range[i-1]
    if d < diff
      diff = d
      result = [p_in_range[i-1], p_in_range[i]]
    end
  end
  result
end
```
