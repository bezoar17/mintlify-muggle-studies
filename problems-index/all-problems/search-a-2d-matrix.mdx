# @param {Integer[][]} matrix
# @param {Integer} target
# @return {Boolean}
def search_matrix(matrix, target)
  m = matrix.size; n = matrix[0].size;
  row_val = 0; # decide the row

  return false if target < matrix[0][0]
  
  while row_val.between?(0, m-1)
    break if target <= matrix[row_val][n-1]
    row_val+=1
  end

  return false if row_val > m-1

  lp = 0; rp = n-1;

  while rp>=lp
    mid = lp + (rp-lp)/2
    return true if target == matrix[row_val][mid]
    if target > matrix[row_val][mid]
      lp = mid+1
    else
      rp = mid-1
    end
  end

  false
end
