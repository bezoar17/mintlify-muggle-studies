class Solution {
public:
    bool isValid(int n,int positions[],int row,int column){
		for(int i=0;i<column;i++)
			if(positions[i]==row || abs(positions[i]-row)== abs(i-column))
				return false;
		return true;
	}
	
	int solveNQ(int n,int column,int positions[],int count){

		if(column==n)
			return count+1;
		for(int i=0;i<n;i++)
			if(isValid(n,positions,i,column)){
				positions[column]=i;
				count=solveNQ(n,column+1,positions,count);
		}
		return count;
	}
	
    int totalNQueens(int n) {
        int positions[n];
       return solveNQ(n,0,positions,0);
    }
};
