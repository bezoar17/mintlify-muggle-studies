```ruby title="Minimum Positive Sum Subarray " lines
# @param {Integer[]} nums
# @param {Integer} l
# @param {Integer} r
# @return {Integer}
class PseudoSumSegTree
  def initialize(arr)
    @n = arr.size # size of the source array
    # query can have r till n, so sum of all elements is at prefix[n]
    @prefix_sum_arr = (1..@n).reduce([0]) {|res, i| res << res.last + arr[i-1]; res } # sum till i-1
  end
  
  # [l,r)
  def query(l,r)
    @prefix_sum_arr[r] - @prefix_sum_arr[l]
  end
end

def minimum_sum_subarray(nums, l, r)
  seg = PseudoSumSegTree.new(nums)
  n = nums.size
  result = Float::INFINITY
  (l..r).each do |x|
    (0...n).each do |i|
      next if i+x>n
      sum = seg.query(i, i+x)
      result = [result, sum].min if sum > 0
    end
  end
  result == Float::INFINITY ? -1:result
end
```
