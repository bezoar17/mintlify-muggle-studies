class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        map<int,int> hash;bool flag=0;
        for(int i=0;i<nums.size();i++){
            if(hash.find(nums[i])==hash.end())
                hash[nums[i]]=i;
            else
            { 
                if(i-hash[nums[i]] <=k)
                flag=1;
                hash[nums[i]]=i;
            }
            
        }
        return flag;
    }
};
