def common_chars(a)
  hash_ = a[0].each_char.inject(Hash.new(0)) {|h,i| h[i]+=1;h}    
  a[1, a.size].each do |str|
    h_ = str.each_char.inject(Hash.new(0)) {|h,i| h[i]+=1;h}
    (hash_.keys - h_.keys).each {|i| hash_.delete(i)}
    h_.keys.each do |key, value|
      hash_[key]= [ hash_[key], h_[key]].min
    end
  end
  result = [] 
  hash_.each do |key, value|
      result += [key] * value
  end
  result
end
