def count_of_substrings(word, k)
    is_vowel = Array.new(128, false)
    freq = Array.new(128, 0)
    vowels = "aeiou"

    # Mark vowels in the is_vowel array
    vowels.each_char { |v| is_vowel[v.ord] = true }

    response = 0
    current_k = 0
    vowel_count = 0
    extra_left = 0
    left = 0

    (0...word.length).each do |right|
        right_char = word[right].ord

        if is_vowel[right_char]
            freq[right_char] += 1
            vowel_count += 1 if freq[right_char] == 1
        else
            current_k += 1
        end

        # Shrink the window if consonant count exceeds k
        while current_k > k
            left_char = word[left].ord
            if is_vowel[left_char]
                freq[left_char] -= 1
                vowel_count -= 1 if freq[left_char] == 0
            else
                current_k -= 1
            end
            left += 1
            extra_left = 0
        end

        # Adjust left pointer to remove extra vowels
        while vowel_count == 5 && current_k == k && left < right && is_vowel[word[left].ord] && freq[word[left].ord] > 1
            extra_left += 1
            freq[word[left].ord] -= 1
            left += 1
        end

        # Count valid substrings
        response += (1 + extra_left) if current_k == k && vowel_count == 5
    end

    response
end
