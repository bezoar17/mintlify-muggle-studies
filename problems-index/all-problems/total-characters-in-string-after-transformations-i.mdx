class Solution:
    def lengthAfterTransformations(self, s: str, t: int) -> int:
        mod = 100_000_000_7
        char_count = [0] * 26
        for ch in s:
            char_count[ord(ch) - ord('a')] += 1

        for _ in range(t):
            next_char_count = [0] * 26
            for i in range(26):
                next_char_count[i] = char_count[i - 1] if i > 0 else 0
            next_char_count[1] += char_count[-1]
            next_char_count[0] += char_count[-1]

            char_count = next_char_count

        return sum(char_count) % mod
