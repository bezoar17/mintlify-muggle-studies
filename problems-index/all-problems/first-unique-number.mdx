  <Tab title="Solution" icon="lightbulb" iconType="regular" >
    <Card>
      Maintain a queue and freq to check for unique elements.
    </Card>
  <Expandable title="code">
    <CodeGroup dropdown>
```ruby title="First Unique Number" lines
class FirstUnique

=begin
    :type nums: Integer[]
=end
  def initialize(nums)
    @freq = Hash.new(0)
    @queue = []
    nums.each {|num| add(num) }
  end


=begin
    :rtype: Integer
=end
  def show_first_unique()
    @queue.shift while @freq[@queue.first] > 1
    @queue.first || -1
  end


=begin
    :type value: Integer
    :rtype: Void
=end
  def add(value)
    if !@freq.include?(value)
      @queue << value
      @freq[value] = 1
    else
      @queue.shift if @queue.first == value
      @freq[value] += 1
    end
  end

end
```
